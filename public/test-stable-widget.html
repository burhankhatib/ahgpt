    usePopupAuth: true,
    fallbackToGuest: true,
    onReady: () => console.log('Widget ready!'),
    onUserSignIn: (user) => console.log('User signed in:', user),
    onUserSignOut: () => console.log('User signed out'),
    onError: (error) => console.error('Widget error:', error)
  });
